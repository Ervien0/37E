CREATE TABLE Requests (
  Id INT IDENTITY(1,1) NOT NULL PRIMARY KEY, -- MySQL: AUTO_INCREMENT
  ClientName VARCHAR(100) NOT NULL,
  RequestType VARCHAR(100) NOT NULL,
  Description TEXT,
  Status VARCHAR(50) NOT NULL,
  CreatedAt DATETIME NOT NULL DEFAULT GETDATE() -- MySQL: CURRENT_TIMESTAMP
);
